struct ListNode* ReversList( struct ListNode *pHead ){
if(pHead==NULL || pHead->next==NULL)return pHead;
struct ListNode *f,*b;
b=pHead->next;f=pHead;f->next=NULL;
while(b){
f=b;
b=b->next;
f->next=pHead;
pHead=f;
}
}
//         f     b
//[NULL]<-[O]-X-[O]
